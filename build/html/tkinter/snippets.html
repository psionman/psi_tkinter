    def _get_data_directory(self):
        """Set data directory."""
        data_directory = filedialog.askdirectory(
            initialdir=Path(self.parent.data_directory.get()).parent,
            parent=self,
        )
        if data_directory:
            self.parent.data_directory.set(data_directory)
            self.parent.config.data_directory = data_directory

    def _get_email_template(self):
        """Set email template path."""
        email_template = filedialog.askopenfilename(
            initialdir=Path(self.parent.email_template.get()).parent,
            initialfile=self.parent.email_template.get(),
            filetypes=TXT_FILE_TYPES,
            parent=self,
        )
        if email_template:
            self.parent.email_template.set(email_template)
            self.parent.config.email_template = email_template
